a = int(input())
x = a%10
y = (a//10)
if (x == 0 or y == 0 ):
print("YES")
else:
ex_1= y/x
ex_2= x/y
res1 = ex_1 - M.floor(ex_1)
res2 = ex_2 - M.floor(ex_2)
if res1 == 0 or res2 == 0:
print("YES")
else:
print("NO")
